# Perceptron

class Perceptron:
    def __init__(self, input_dim, eta=1.0):
        self.w = np.zeros(input_dim)
        self.b = 0
        self.eta = eta
    
    def step(self, x):
        return 1*(x > 0)
    
    def y(self, w, x):
        return self.step(np.dot(w, x) + self.b)
    
    def fit(self, X, Y):
        while True:
            classified = True
            for i in range(20):
                update = self.eta*(Y[i] - self.y(self.w, X[i]))
                delta_w = update * X[i]
                delta_b = update
                self.w += delta_w
                self.b += delta_b
                classified *= all(delta_w == 0)*(delta_b == 0)
            if classified:
                break
        return 

pt = Perceptron(input_dim = 2)
pt.fit(X,Y)
print('w = ', pt.w)
print('b = ', pt.b)
